#include <bits/stdc++.h> 
int findMinimumCoins(int amount) 
{
    // Write your code here
    int coins[] = {1,2,5,10,20,50,100,500,1000};
    int minCoins=0;
    for(int i=8;i>=0;i--){
        if(amount-coins[i]>=0){
            while(amount-coins[i]>=0){
                amount=amount-coins[i];
                minCoins++;
            }
        }
        if(amount==0)break;
    }
    return minCoins;
}
